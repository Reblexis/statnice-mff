\ulohahead{společná informatika \textnormal{\footnotesize[úroveň 2]}}{Stránkování, překlad adresy a ochrana paměti}
\begin{zadani}
16bitová virtuální adresa, velikost stránky 256\,B. Virtuální adresa je \texttt{0x1234}.
\begin{enumerate}[label=(\alph*)]
  \item \bod{1} Rozdělte adresu na číslo stránky a offset.
  \item \bod{2} Jednoúrovňová stránkovací tabulka mapuje stránku \texttt{0x12} na rámec \texttt{0x07}. Spočtěte fyzickou adresu.
  \item \bod{3} Vysvětlete roli virtuálních adresových prostoru v ochraně paměti procesu.
\end{enumerate}
\end{zadani}

\begin{reseni}
\textbf{(a)} Stránka 256\,B $\Rightarrow$ offset má $\log_2 256=8$ bitu. \texttt{0x1234} $=$ \texttt{0001\,0010\,0011\,0100}$_2$: offset (dolních 8 bitu) $=$ \texttt{0x34}, číslo stránky (horních 8 bitu) $=$ \texttt{0x12}.

\textbf{(b)} Stránka \texttt{0x12} $\to$ rámec \texttt{0x07}. Fyzická adresa $=$ (rámec $\ll 8$) $\,|\,$ offset $=$ \texttt{0x07}\,$\ll8\ |\ $\texttt{0x34} $=$ \texttt{0x0734}.

\textbf{(c)} Každý proces má vlastní stránkovací tabulku, tedy vlastní mapování virtuálních adres na rámce; nevidí proto fyzickou paměť jiných procesu (\emph{izolace}). Přístup na nenamapovanou stránku vyvolá výpadek (page fault) a OS proces ochrání/ukončí. Tím virtuální adresový prostor zajišťuje ochranu paměti mezi procesy.

\emph{(Nad rámec požadavku - jen pro přehled:) u velkých adresových prostoru se používají \emph{víceúrovňové} tabulky (alokují podtabulky jen pro použité oblasti, šetří paměť) a \emph{TLB} (cache nedávných překladu stránka$\to$rámec).}
\end{reseni}

\begin{jadro}
Adresa = (číslo stránky $\Vert$ offset); offset má $\log_2$(velikost stránky) bitu. Fyzická $=$ rámec $\ll$ (bity offsetu) $|$ offset. Virtuální adresový prostor (vlastní tabulka na proces) zajišťuje izolaci/ochranu paměti.
\end{jadro}

\kalibrace{Překlad adres jednoúrovňovou tabulkou + ochrana paměti jsou přesně v požadavku a opakovaně se zkouší (virtuální paměť, architektura/OS ~60\,\%). Rozdělení adresy + výpočet rámce = 2-3 body.}
\past{Offset má tolik bitu, kolik je $\log_2$(velikost stránky). Fyzická adresa skládá \emph{rámec} (ne číslo stránky) s offsetem.}
