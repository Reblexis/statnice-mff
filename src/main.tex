\section{Co a jak hodně se učit (z rozboru 20 zkoušek 2019-2026)}

\begin{infobox}[title={Hlavní zjištění z reálných zadání}]
Tohle je výtah z rozboru 20 minulých zadání (\texttt{zkousky/}, audit v
\texttt{COMPLETENESS-EXAMS.md}). Říká, kam dát čas, aby na jistou trojku stačilo co nejméň učení.
\end{infobox}

\begin{itemize}[leftmargin=1.3em]
  \item \textbf{Anatomie otázky (podle vzorových řešení):} každá otázka má 3 podotázky a body
  se dávají po částech. (1) \emph{přesná definice / znění věty / algoritmu}, (2) konstrukce,
  pseudokód nebo malý výpočet, (3) okrajový případ, složitost nebo aplikace.
  \textbf{Vždy si zajisti bod (1)}, je nejjistější a nejlevnější.
  \item \textbf{Definice se zkoušejí nejvíc} (slovo „Definujte“ je v zadáních 68$\times$). Uměj
  \emph{přesné} definice zpaměti; to je největší výnos za nejmíň času. Proto je v každém tématu
  \bod{1} přímo přesná definice.
  \item \textbf{Specializace (UI-SU) je konceptuální.} Vzorová řešení říkají „popište“,
  „vysvětlete“, \emph{„nemusíte nic počítat“}, „zakreslete přibližně“. Uč se \emph{principy} a
  uměj je vysvětlit a načrtnout; jen pár míst chce malý výpočet (jedna aktualizace EM, metriky
  z matice záměn). Sem dej \textbf{většinu času} (potřebuješ 7/12).
  \item \textbf{Společná matematika a informatika: definice + jeden výpočet.} Vzorová řešení mají
  i plné výpočty (integrál, diagonalizace, sestrojení automatu, kód), ale stačí 5/12, takže jeď
  na šířku: definice u všech témat + natrénuj jeden typový výpočet u nejčastějších.
  \item \textbf{Kolik detailu:} úroveň „Nástinu řešení“ v zadáních = přesná definice + hlavní
  postup/algoritmus + jeden příklad. Dlouhé formální důkazy nemusíš umět do detailu, ale
  \textbf{otázka se může týkat i důkazu} - uměj aspoň \emph{hlavní myšlenku / náčrt} u klíčových vět.
\end{itemize}

\begin{infobox}[title={Nejčastější (= nejvýnosnější) okruhy}]
\begin{itemize}[leftmargin=1.2em]
  \item \textbf{Spol. informatika:} automaty (regulární, CFG/PDA, Chomského hierarchie),
  grafové algoritmy (BFS/DFS, topologické třídění, kostry, toky), haldy a vyhledávací stromy,
  třídění; OS (paměť, virtuální paměť, synchronizace, zařízení, reprezentace dat).
  \item \textbf{Spol. matematika:} lineární algebra, jednorozměrná analýza, základy grafů,
  pravděpodobnost a teorie informace, definice ve výrokové/predikátové logice.
  \item \textbf{Specializace UI-SU:} evaluační metriky, regrese, shlukování, rozhodovací stromy,
  ensembly, SVM/PCA, Bayesovské sítě/HMM/EM, CSP a prohledávání, hry/MDP.
\end{itemize}
\end{infobox}

\noindent\textbf{Pozn.\ k formátu:} zadání 2019-2022 jsou starší (specializace Strojové učení
ještě neexistovala), proto z nich platí jen společné okruhy. Od 2023 je dělení čisté:
2 matematika + 2 informatika + 4 specializace (UI-SU).

\section{Strategie: jak projít s minimem času}

\begin{infobox}[title={Bodový rozpočet (proč to stačí)}]
8 otázek po 0-3 bodech. \textbf{Průchod (známka 3)} = aspoň 12 bodů celkem \textbf{a} aspoň
5 ze společných okruhů (4 otázky) \textbf{a} aspoň 7 ze specializace (4 otázky).
\end{infobox}

\begin{infobox}[title={Logistika zkoušky (dle pokynů k termínu)}]
\begin{itemize}[leftmargin=1.2em]
  \item \textbf{Čas:} místnost se otevírá 8:45, začátek 9:00, psaní až do 16:00 (7 hodin).
  Práce je zhruba na 2-3 h, takže \textbf{časový tlak je minimální} - v klidu zkus všech 8 otázek.
  \item \textbf{Otázky:} 8 (2 matematika, 2 informatika, 4 specializace), mohou se dělit a
  \textbf{týkat se definic, aplikací i důkazů}; libovolné téma z požadavků oboru.
  \item \textbf{Anonymně:} píše se na listy zadání (případně volné listy), značí se
  \emph{číslem otázky a číslem studenta}, ne jménem.
  \item \textbf{S sebou:} doklad totožnosti, psací potřeby, jídlo a pití (dlouhé). \textbf{Zakázané}
  (nesmíš mít ani u sebe): knihy, poznámky, kalkulačka, telefon, tablet; odkládá se u oken.
  \item \textbf{Výsledky:} mailem cca 2 pracovní dny po zkoušce, s ohodnocenými odpověďmi; o revizi
  lze požádat do 1 pracovního dne. Anglická varianta na vyžádání do 1 pracovního dne (závazné).
\end{itemize}
\end{infobox}

\begin{itemize}[leftmargin=1.3em]
  \item \textbf{Specializace je klíčová} (potřebuješ 7/12, tj.\ průměr 1,75 na otázku). Sem dej
  většinu času (ML + Základy UI $\approx$ 60 \%). Společné okruhy mají nízký floor 5/12
  ($\approx$ 1,25 na otázku) - stačí breadth, ne hloubka.
  \item \textbf{Hlavní riziko je nula z přidělené otázky}, ne chybějící důkaz. Proto měj u
  \emph{každého} tématu aspoň „kostru na 1 bod“. Žádné nuly $\Rightarrow$ pravděpodobnost
  průchodu jde nad 98 \%.
  \item \textbf{Trénuj nad hranici} (cíl $\sim$16/24), protože 5+7=12 nemá rezervu.
  \item \textbf{Vynech} (bezpečně): plná formální odvození (ale hlavní myšlenku důkazu klíčových
  vět uměj, důkaz se může objevit), jazykové varianty mimo zvolený jazyk, vše mimo rozsah.
  Detaily viz sekce \ref{sec:rozsah}.
  \item \textbf{V samotné zkoušce máš spoustu času} (7 h na zhruba 2-3 h práce). Nehoň se, zkus
  \emph{všech 8 otázek} a u každé aspoň definici; rychlost není faktor.
\end{itemize}

\subsection{Kolik času to zabere (kalibrovaný odhad)}
{\small Odhady jsou pro studenta, který předměty \textbf{absolvoval} (jde o opakování z tohoto
textu). Podle svého základu klidně uprav o $\pm$50~\%. „Žlutá“ = umíš definice (síť proti
nulám); „zelená“ = navíc algoritmus/postup + příklad (cíl pro 2 body). Časy na zelenou jsou
kumulativní (zahrnují žlutou).}

\begin{center}\small
\begin{tabular}{@{}lccc@{}}
\toprule
\textbf{Okruh} & \textbf{témat} & \textbf{na žlutou} & \textbf{na zelenou} \\
\midrule
Společná matematika & 6 & $\approx$4 h & $\approx$10 h \\
Společná informatika & 4 & $\approx$4 h & $\approx$12 h \\
Základy UI & 9 & $\approx$5 h & $\approx$13 h \\
Strojové učení & 9 & $\approx$5 h & $\approx$14 h \\
Vzorové výpočty + 2-3 staré zkoušky na čas & - & - & $\approx$8 h \\
\bottomrule
\end{tabular}
\end{center}

\noindent Z toho plyne přibližná křivka \textbf{celkový čas $\to$ pravděpodobnost složení}
(známka 3+; předpokládá rozumné využití času podle priorit, tj.\ nejdřív specializace):

\begin{center}\small
\begin{tabular}{@{}cll@{}}
\toprule
\textbf{čas} & \textbf{$\approx$P(složení)} & \textbf{co za ten čas zvládneš} \\
\midrule
$\approx$18 h & $\approx$55-60 \% & definice všude (žlutá), specializaci rozkoukáno \\
$\approx$30 h & $\approx$78 \% & specializace zelená + společné žluté (na hranici floorů) \\
$\approx$42 h & $\approx$88 \% & + společné téměř zelené, pár drilů \\
$\approx$55 h & $\approx$93-95 \% & vše zelené + vzorové výpočty + 2-3 staré zkoušky \\
$\approx$65 h & $\approx$96 \% & + víc procvičených zkoušek a kódu rukou \\
\bottomrule
\end{tabular}
\end{center}

\noindent\textbf{Největší výnos na hodinu:} dostat specializaci (ML + Základy UI) na zelenou a
\textbf{udělat staré zkoušky na čas}. Naopak honit 3.\ bod u už zeleného tématu je nejmenší výnos.

\subsection{Rubrika zvládnutí (sebehodnocení)}
\begin{center}\small
\begin{tabular}{@{}cl@{}}
\toprule
\textbf{stav} & \textbf{co umíš} \\
\midrule
\textcolor{gapred}{$\bullet$}\ červená (0) & ani definici \\
\textcolor{orange!80!black}{$\bullet$}\ žlutá (1) & definice + pár pojmů \\
\textcolor{intugreen}{$\bullet$}\ zelená (2) & definice + algoritmus/věta + příklad + intuice \quad\textbf{(cíl)} \\

\end{center}
\noindent Cíl: každé téma aspoň žlutě; specializace $\sim$85 \% zeleně; nejdůležitější modely modře.
