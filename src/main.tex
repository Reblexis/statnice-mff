\section{Strategie: jak projít s minimem času}

\begin{infobox}[title={Bodový rozpočet (proč to stačí)}]
8 otázek po 0-3 bodech. \textbf{Průchod (známka 3)} = aspoň 12 bodů celkem \textbf{a} aspoň
5 ze společných okruhů (4 otázky) \textbf{a} aspoň 7 ze specializace (4 otázky).
\end{infobox}

\begin{itemize}[leftmargin=1.3em]
  \item \textbf{Specializace je klíčová} (potřebuješ 7/12, tj.\ průměr 1,75 na otázku). Sem dej
  většinu času (ML + Základy UI $\approx$ 60 \%). Společné okruhy mají nízký floor 5/12
  ($\approx$ 1,25 na otázku) - stačí breadth, ne hloubka.
  \item \textbf{Hlavní riziko je nula z přidělené otázky}, ne chybějící důkaz. Proto měj u
  \emph{každého} tématu aspoň „kostru na 1 bod“. Žádné nuly $\Rightarrow$ pravděpodobnost
  průchodu jde nad 98 \%.
  \item \textbf{Trénuj nad hranici} (cíl $\sim$16/24), protože 5+7=12 nemá rezervu.
  \item \textbf{Vynech} (bezpečně): dlouhé důkazy, plná odvození, jazykové varianty mimo zvolený
  jazyk, vše mimo rozsah. Detaily viz sekce \ref{sec:rozsah}.
\end{itemize}
